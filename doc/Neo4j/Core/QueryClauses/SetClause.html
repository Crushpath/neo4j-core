def from_key_and_value(key, value)
  case value
  when String, Symbol
    "#{key} = #{value}"
  when Hash
    if @options[:set_props]
      attribute_string = value.map { |k, v| "#{k}: #{v.inspect}" }.join(', ')
      "#{key} = {#{attribute_string}}"
    else
      value.map do |k, v|
        key_value_string("#{key}.`#{k}`", v, ['setter'], true)
      end
    end
  else
    fail ArgError, value
  end
end
